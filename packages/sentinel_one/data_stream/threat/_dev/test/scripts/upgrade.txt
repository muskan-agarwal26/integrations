# Test that upgrading from 2.8.2 (declarative httpjson) to ${CURRENT_VERSION}
# (run_as_cel) preserves the threat stream cursor and continues incremental
# collection.
#
# 2.8.2 is hardcoded the last EPR release before the run_as_cel migration —
# not ${PREVIOUS_VERSION}, which would drift on every package bump.
#
# Phase 1 (2.8.2, declarative httpjson): collect 2 threats with updatedAt
# 2024-01-15T10:01Z and 2024-01-15T10:06Z. Cursor should advance to the last
# updatedAt (10:06Z).
#
# After upgrade_package_latest, phase 2 mock returns the new threat only when
# updatedAt__gte matches a preserved httpjson cursor (2024-01-15T10:01* or
# 2024-01-15T10:06*). A reset to initial_interval uses a 2026-* lookback and
# does not match a rule, yielding no new data.
#
# Expected outcome when cursor is preserved: 3 unique threat documents total.
# Expected failure when cursor resets: count stays at 2 (pipeline fingerprint
# deduplicates re-fetched threats, so duplicate doc inflation is not observable).

[!external_stack] skip 'Skipping external stack test.'
[!exec:jq] skip 'Skipping test requiring absent jq command'
[breaking_change] skip 'Cannot upgrade across breaking change.'

use_stack -profile ${CONFIG_PROFILES}/${PROFILE}
install_agent -profile ${CONFIG_PROFILES}/${PROFILE} -network_name NETWORK_NAME
docker_up -profile ${CONFIG_PROFILES}/${PROFILE} -network ${NETWORK_NAME} s1-mock

install_package_from_registry -profile ${CONFIG_PROFILES}/${PROFILE} ${PACKAGE_NAME} 2.8.2
add_package_policy -profile ${CONFIG_PROFILES}/${PROFILE} -version 2.8.2 test_config.yaml DATA_STREAM_NAME

get_docs -profile ${CONFIG_PROFILES}/${PROFILE} -want 2 -confirm 15s -timeout 5m ${DATA_STREAM_NAME}
cp stdout got_docs_pre_upgrade.json

exec jq '[.hits.hits[]._source.sentinel_one.threat.threat_id] | sort | join(",")' got_docs_pre_upgrade.json
stdout '"threat-00000000000000000000000000000001,threat-00000000000000000000000000000002"'

add_package -profile ${CONFIG_PROFILES}/${PROFILE}

upgrade_package_latest -profile ${CONFIG_PROFILES}/${PROFILE}
stdout 'upgraded package '${PACKAGE_NAME}

docker_down s1-mock
cp s1-mock/config-phase2.yml s1-mock/config.yml
docker_up -profile ${CONFIG_PROFILES}/${PROFILE} -network ${NETWORK_NAME} s1-mock

get_docs -profile ${CONFIG_PROFILES}/${PROFILE} -want 3 -confirm 15s -timeout 5m ${DATA_STREAM_NAME}
cp stdout got_docs_post_upgrade.json

exec jq '[.hits.hits[]._source.sentinel_one.threat.threat_id | select(. != null)] | unique | sort | join(",")' got_docs_post_upgrade.json
stdout '"threat-00000000000000000000000000000001,threat-00000000000000000000000000000002,threat-00000000000000000000000000000003"'

exec jq '[.hits.hits[]._source | select(.sentinel_one.threat.threat_id != null)] | length' got_docs_post_upgrade.json
stdout '^3$'

remove_package_policy -profile ${CONFIG_PROFILES}/${PROFILE} ${DATA_STREAM_NAME}
uninstall_agent -profile ${CONFIG_PROFILES}/${PROFILE} -timeout 1m
docker_down s1-mock
